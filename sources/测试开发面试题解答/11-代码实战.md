## Q33：手写一个 requests 请求封装

要求：支持 GET/POST、headers/params/data/json、超时、日志、异常捕获、统一返回、敏感字段脱敏、扩展 PUT/DELETE、失败给清晰错误。

```python
import logging
import re
import requests
from requests.exceptions import RequestException, Timeout, ConnectionError

log = logging.getLogger(__name__)

# 统一用小写比较，兼容 Authorization / access_token 等各种写法
SENSITIVE_KEYS = {"password", "token", "authorization",
                  "access_token", "refresh_token", "idcard", "phone"}


def _mask(value):
    if isinstance(value, str) and len(value) > 6:
        return value[:3] + "****"
    return "****"


def _desensitize(data):
    """递归脱敏：处理 dict / list 任意层级嵌套，key 大小写不敏感"""
    if isinstance(data, dict):
        return {k: (_mask(v) if k.lower() in SENSITIVE_KEYS else _desensitize(v))
                for k, v in data.items()}
    if isinstance(data, list):
        return [_desensitize(i) for i in data]
    return data


def _mask_text(text):
    """兜底：对响应/表单等文本里的敏感字段做正则脱敏，避免明文进日志"""
    if not isinstance(text, str):
        return text
    keys = "|".join(SENSITIVE_KEYS)
    return re.sub(rf'("(?:{keys})"\s*:\s*")([^"]+)',
                  lambda m: m.group(1) + m.group(2)[:3] + "****",
                  text, flags=re.IGNORECASE)


class ApiResponse:
    """统一响应封装"""
    def __init__(self, resp: requests.Response):
        self.raw = resp
        self.status_code = resp.status_code
        try:
            self.body = resp.json()
        except ValueError:
            self.body = resp.text

    def get(self, path, default=None):
        """支持 'data.user.id' 这种点路径取嵌套字段"""
        node = self.body
        for key in path.split("."):
            if isinstance(node, dict) and key in node:
                node = node[key]
            else:
                return default
        return node


RETRYABLE_STATUS = {502, 503, 504}
# 幂等方法才允许自动重试；POST/PATCH 默认不重试，避免重复下单/支付
IDEMPOTENT_METHODS = {"GET", "PUT", "DELETE", "HEAD", "OPTIONS"}


class HttpClient:
    def __init__(self, base_url="", timeout=10, retry=2, auth_provider=None):
        self.base_url = base_url.rstrip("/")
        self.timeout = timeout
        self.retry = retry
        self.auth_provider = auth_provider   # 可调用对象，每次请求返回最新 token
        self.session = requests.Session()

    def _build_url(self, url):
        if url.startswith("http"):
            return url
        return f"{self.base_url}/{url.lstrip('/')}"   # 兼容 'api/x' 与 '/api/x'

    def request(self, method, url, *, params=None, data=None, json=None,
                headers=None, files=None, timeout=None, retry=None):
        method = method.upper()
        url = self._build_url(url)
        timeout = timeout or self.timeout
        # 非幂等方法默认不重试；幂等方法用默认次数；也可显式覆盖
        if retry is None:
            retry = self.retry if method in IDEMPOTENT_METHODS else 0

        # 鉴权统一在请求层注入，保证每次都拿最新（可能已刷新）的 token
        headers = dict(headers or {})
        if self.auth_provider and "Authorization" not in headers:
            headers["Authorization"] = f"Bearer {self.auth_provider()}"

        log.info(f"[请求] {method} {url} | params={_desensitize(params)} "
                 f"| data={_desensitize(data)} | json={_desensitize(json)} "
                 f"| headers={_desensitize(headers)}")

        last_exc = None
        for i in range(retry + 1):
            try:
                resp = self.session.request(
                    method=method, url=url, params=params,
                    data=data, json=json, headers=headers,
                    files=files, timeout=timeout)
                # 网关类状态码且仍有重试次数时重试（retry=0 的非幂等请求不会进入）
                if resp.status_code in RETRYABLE_STATUS and i < retry:
                    log.warning(f"[重试] 第 {i + 1} 次返回 {resp.status_code}，重试")
                    continue
                log.info(f"[响应] {resp.status_code} | {_mask_text(resp.text[:500])}")
                return ApiResponse(resp)
            except (ConnectionError, Timeout) as e:      # 网络类异常重试
                last_exc = e
                log.warning(f"[重试] 第 {i + 1} 次网络异常：{e}")
            except RequestException as e:                # 其他请求异常不重试
                log.error(f"[请求异常] {method} {url}: {e}")
                raise RuntimeError(f"请求异常：{method} {url}，原因：{e}") from e
        raise RuntimeError(f"请求失败，重试 {retry} 次仍异常：{method} {url}，"
                           f"最后错误：{last_exc}")

    # 各方法语法糖，统一走 request
    def get(self, url, **kwargs):
        return self.request("GET", url, **kwargs)

    def post(self, url, **kwargs):
        return self.request("POST", url, **kwargs)

    def put(self, url, **kwargs):
        return self.request("PUT", url, **kwargs)

    def delete(self, url, **kwargs):
        return self.request("DELETE", url, **kwargs)
```

**讲解要点**：方法透传支持任意 HTTP method；`timeout` 必传；`_build_url` 用 `base + lstrip('/')` 兼容 `api/x` 与 `/api/x` 两种写法，避免拼成 `comapi/x`；**请求（params/data/json/headers）和响应都走递归脱敏、key 大小写兼容**，响应文本再用 `_mask_text` 正则兜底，避免 token 明文进日志；**重试只对幂等方法或显式开启时生效，覆盖网络异常 + 502/503/504 网关状态码，非幂等（POST/PATCH）默认不重试**，避免重复下单；`ApiResponse` 提供点路径取值给断言层；失败抛 `RuntimeError` 并带 method/url/原因。

## Q34：手写一个 session 级 fixture，登录获取 token

要求：session 级；调登录接口；成功返回带 token 的 headers；失败抛异常；用例可直接注入；token 过期支持重新获取；日志不打印完整 token。

```python
import time
import pytest
from common.http_client import HttpClient


class TokenManager:
    """带过期判断的 token 管理，支持自动刷新"""
    def __init__(self, base_url, username, password, ttl=1800):
        self.client = HttpClient(base_url=base_url)
        self.username = username
        self.password = password
        self.ttl = ttl                 # token 有效期（秒）
        self._token = None
        self._expire_at = 0

    def _login(self):
        resp = self.client.post("/api/login",
                                json={"username": self.username,
                                      "password": self.password})
        if resp.status_code != 200 or resp.get("code") != 0:
            raise RuntimeError(f"登录失败，无法获取 token：{resp.body}")
        token = resp.get("data.token")
        assert token, "登录响应未返回 token"
        return token

    def get_token(self, monotonic=time.monotonic):
        """token 不存在或已过期时重新登录"""
        if self._token is None or monotonic() >= self._expire_at:
            self._token = self._login()
            self._expire_at = monotonic() + self.ttl
        return self._token


@pytest.fixture(scope="session")
def token_manager():
    return TokenManager(base_url="https://api.example.com",
                        username="test01", password="123456")


@pytest.fixture(scope="session")
def client(token_manager):
    """鉴权下沉到请求层：每次发请求时才调 get_token()，
    token 过期会自动重登，用例完全无感知。"""
    return HttpClient(base_url="https://api.example.com",
                      auth_provider=token_manager.get_token)
```

用例直接注入：

```python
def test_create_order(client):
    resp = client.post("/api/order", json={"sku": "1001", "num": 1})
    assert resp.get("code") == 0
```

**讲解要点**：

- `TokenManager` 与 `token_manager` 为 **session 级**，全局只在需要时登录，过期才重登（要点 1、6）。
- **关键在"token 在哪一层注入"**：本例放在 **请求层**（`auth_provider`，每次发请求都调 `get_token()`），所以 token 刷新后下一次请求立即用上新值。
- ⚠️ **反例（容易被追问穿）**：若写成 `login_headers` 返回一个 **session 级静态 dict**——`return {"Authorization": f"Bearer {token_manager.get_token()}"}`——这个 header 只在 session 初始化时算 **一次**，之后即使 `get_token()` 内部刷新了 token，用例拿到的仍是 **旧 header**，"失效自动重登"就成了空话。正确做法：把鉴权放请求层（推荐），或让 `login_headers` 降为 function 级、或返回一个"每次动态取 header"的方法。
- 登录失败主动抛异常而非返回 None（要点 4）；token 脱敏交给 HttpClient（要点 7）。

## Q35：手写一个 YAML 数据驱动接口用例

要求：YAML 存接口名/地址/方法/参数/预期；Python 读 YAML；pytest 参数化；按预期断言 code；Allure 动态标题；字段缺失给清晰错误；支持多组数据。

```yaml
# data/login.yaml
- name: 登录成功
  url: /api/login
  method: POST
  json: {username: test01, password: "123456"}
  expected: {code: 0}

- name: 密码错误
  url: /api/login
  method: POST
  json: {username: test01, password: "wrong"}
  expected: {code: 10001}
```

```python
import yaml
import pytest
import allure
from common.http_client import HttpClient

REQUIRED = ("name", "url", "method", "expected")


def load_cases(path):
    try:
        with open(path, encoding="utf-8") as f:
            cases = yaml.safe_load(f)
    except FileNotFoundError:
        raise FileNotFoundError(f"用例文件不存在：{path}")
    except yaml.YAMLError as e:
        raise ValueError(f"YAML 解析失败 {path}：{e}")

    if not isinstance(cases, list):
        raise ValueError(f"{path} 顶层应为用例列表")
    for i, c in enumerate(cases, 1):
        missing = [k for k in REQUIRED if k not in c]
        if missing:
            raise ValueError(f"{path} 第 {i} 条用例缺少字段：{missing}")
    return cases


cases = load_cases("data/login.yaml")


@pytest.mark.parametrize("case", cases, ids=[c["name"] for c in cases])
def test_login(case):
    allure.dynamic.title(case["name"])           # Allure 显示中文用例标题
    client = HttpClient(base_url="https://api.example.com")
    resp = client.request(case["method"], case["url"], json=case.get("json"))
    expected_code = case["expected"]["code"]
    assert resp.get("code") == expected_code, \
        f"用例[{case['name']}] 期望 code={expected_code}，实际={resp.get('code')}"
```

**讲解要点**：`load_cases` 在读取阶段就做缺失字段校验并报清晰错误（要点 6）；`ids` + `allure.dynamic.title` 让报告显示中文标题（要点 5）；list 多元素天然支持多组数据（要点 7）。

## Q36：手写动态参数提取和替换逻辑

要求：从响应 JSON 提取 `data.orderId`；存上下文；后续接口用 `${orderId}`；请求前自动替换；字段不存在给清晰错误；支持嵌套；支持 token/orderId/userId 多变量；并发避免数据互相污染。

```python
import re
import threading


class Context:
    """用例间共享的运行时上下文（线程安全）"""
    def __init__(self):
        self._data = {}
        self._lock = threading.Lock()

    def save(self, key, value):
        with self._lock:
            self._data[key] = value

    def get(self, key):
        with self._lock:
            if key not in self._data:
                raise KeyError(f"上下文中不存在变量：{key}")
            return self._data[key]

    def replace(self, obj):
        """递归替换 ${var} 占位符，支持 dict / list / str 嵌套"""
        if isinstance(obj, str):
            for name in re.findall(r"\$\{(\w+)\}", obj):
                obj = obj.replace(f"${{{name}}}", str(self.get(name)))
            return obj
        if isinstance(obj, dict):
            return {k: self.replace(v) for k, v in obj.items()}
        if isinstance(obj, list):
            return [self.replace(v) for v in obj]
        return obj


def extract(body, path):
    """从响应中按 'data.orderId' 点路径提取，支持嵌套"""
    node = body
    for key in path.split("."):
        if isinstance(node, dict) and key in node:
            node = node[key]
        else:
            raise KeyError(f"响应中不存在字段：{path}")
    return node
```

使用示例：

```python
ctx = Context()

# 下单后提取 orderId、userId 存上下文
order_resp = OrderApi.create(...)
ctx.save("orderId", extract(order_resp.body, "data.orderId"))
ctx.save("userId", extract(order_resp.body, "data.user.id"))   # 嵌套提取

# 支付请求体里用 ${orderId}，发请求前自动替换
pay_body = ctx.replace({"orderId": "${orderId}", "userId": "${userId}"})
PayApi.pay(json=pay_body)
```

**讲解要点**：

- `extract` 支持 `data.user.id` 这种 **嵌套字段**（要点 6），字段不存在抛清晰 `KeyError`（要点 5）。
- `replace` 递归处理 dict/list/str，支持 token/orderId/userId 等 **任意多变量**（要点 7）。
- **并发数据污染**（要点 8）：每个用例/线程使用 **独立 Context 实例**（不共享）；若必须共享，用 `threading.Lock` 保护读写（上面已加锁），避免多线程同时读写导致脏数据。
